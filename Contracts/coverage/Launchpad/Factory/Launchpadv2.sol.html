    function validateCreateInput(
        LaunchPadLib.PresaleType _presaleType,
        LaunchPadLib.TokenInfo memory _tokenInfo,
        LaunchPadLib.ParticipationCriteria memory _participationCriteria,
        LaunchPadLib.Tokenomics memory _tokenomics,
        LaunchPadLib.PresaleTimes memory _presaleTimes,
        LaunchPadLib.ReqestedTokens memory _reqestedTokens
        ) internal view  {
        
 
        Erequire( address(_tokenInfo.preSaleToken) != address(0), "Presale project address can't be null");
                
        Iif(_presaleType == LaunchPadLib.PresaleType.TOKENHOLDERS) {
            require( _participationCriteria.criteriaToken != address(0), "Criteria token address can't be null");
        }
 
        Erequire( _tokenomics.tokensPCForLP >= 50 && _tokenomics.tokensPCForLP <= 100, "liquidity should be at least 50% or more");
        
        Erequire( _tokenomics.tokensForSale > 0, "tokens for sale must be more than 0");
 
        Erequire( _reqestedTokens.minTokensReq > 0, "_minTokensReq should be more than zero");
        Erequire( _reqestedTokens.maxTokensReq > _reqestedTokens.minTokensReq, "_maxTokensReq > _minTokensReq");
        Erequire( _reqestedTokens.softCap >= _tokenomics.tokensForSale.div(2), "softcap should be at least 50% or more");
 
        Erequire ( _presaleTimes.startedAt >= block.timestamp + 15 minutes, "startedAt should be more than 15 minutes from now" );
        Erequire ( _presaleTimes.expiredAt >= block.timestamp + 1 days, "expiredAt should be more than one day from now" );
        Erequire ( _presaleTimes.lpLockupTime >= block.timestamp + 7 days, "Lockup period should be  7 or more days from now time" );
 
        Erequire ( _participationCriteria.priceOfEachToken > 0, "_priceOfEachToken should be more than zero" );
 
    }
 
    function createPresale(
        LaunchPadLib.TokenInfo memory _tokenInfo,
        LaunchPadLib.Tokenomics memory _tokenomics,
        LaunchPadLib.ParticipationCriteria memory _participationCriteria,
        LaunchPadLib.PresaleTimes memory _presaleTimes,
        LaunchPadLib.ReqestedTokens memory _reqestedTokens
        ) public payable {
 
            validateCreateInput(
                _participationCriteria.typeOfPresale, 
                _tokenInfo, 
                _participationCriteria, 
                _tokenomics,
                _presaleTimes,
                _reqestedTokens
                );
 
        bool exemtFromFee = isUserWhitelistedToStartProject[msg.sender] || msg.sender == owner();
 
        if(!exemtFromFee) {
            Erequire( msg.value >= upfrontfee, "Insufficient funds to start");
        }
 
        presaleCount++;
        
        LaunchPadLib.PresaleInfo memory _presaleInfo = LaunchPadLib.PresaleInfo(
            presaleCount,
            msg.sender,
            LaunchPadLib.PreSaleStatus.PENDING
        );
 
        Presale _presale = new Presale ( 
                salesFeeInPercent,
                _presaleInfo,
                _tokenInfo,
                _participationCriteria,
                _tokenomics,
                _presaleTimes,
                _reqestedTokens,
                launchpadAddresses
         );
 
 
 
        transferTokens(
            _tokenInfo.preSaleToken, 
            address(_presale), 
            _tokenomics.tokensForSale, 
            _tokenomics.tokensPCForLP, 
            _tokenomics.tokensForLocker,
            _tokenInfo.decimals
            );
 
       
        presaleRecordByToken[_tokenInfo.preSaleToken] = address(_presale);
        presaleRecordByID[presaleCount] = address(_presale);
 
 
    }
 
    function transferTokens(address _preSaleToken, address _presale, uint _tokensForSale, uint _reservedTokensPCForLP, uint _tokensForLocker, uint _decimal) internal {
        
        uint tokensForLP = _tokensForSale.mul(_reservedTokensPCForLP).div(100);
        uint totalTokens = _tokensForSale.add(tokensForLP).add(_tokensForLocker);
 
        // console.log("Trying to transfer tokens: ", totalTokens.mul(10**_decimal));
        Erequire(
            IERC20(_preSaleToken).transferFrom(msg.sender, _presale, totalTokens.mul(10**_decimal)),
             "Unable to transfer presale tokens to the contract"
            );
    }
 
    function whiteListUsersToStartProject(address[] memory _addresses) public onlyOwner {
        for(uint i=0; i < _addresses.length; i++){
            isUserWhitelistedToStartProject[_addresses[i]] = true;
        }
    }

        salesFeeInPercent = _salesFeeInPercent;
    }
